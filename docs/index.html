Overview

SmallAPI is a lightweight, zero-dependency Go web framework that brings the simplicity of Python's Flask to Go development. Built entirely on Go's standard library, it provides an intuitive API for creating web applications and APIs with minimal boilerplate.

Hello World in 4 lines:

app := smallapi.New()
app.Get("/", func(c *smallapi.Context) {
    c.JSON(map[string]string{"hello": "world"})
})
app.Run(":8080")

Basic Example

REST API with validation:

package main

import "github.com/grandpaej/smallapi"

type User struct {
    Name  string `json:"name" validate:"required,min=2"`
    Email string `json:"email" validate:"required,email"`
}

func main() {
    app := smallapi.New()
    
    // Middleware
    app.Use(smallapi.Logger())
    app.Use(smallapi.CORS())
    
    // Routes
    app.Get("/users/:id", func(c *smallapi.Context) {
        id := c.Param("id")
        c.JSON(map[string]string{"user_id": id})
    })
    
    app.Post("/users", func(c *smallapi.Context) {
        var user User
        if err := c.JSON(&user); err != nil {
            c.Status(400).JSON(map[string]string{"error": "Invalid JSON"})
            return
        }
        
        if err := c.Validate(&user); err != nil {
            c.Status(400).JSON(map[string]string{"error": err.Error()})
            return
        }
        
        c.Status(201).JSON(user)
    })
    
    app.Run(":8080")
}

Examples

Static Files & Templates:

app := smallapi.New()

// Serve static files
app.Static("/static", "./assets")

// Template rendering
app.Get("/", func(c *smallapi.Context) {
    data := map[string]interface{}{
        "title": "Welcome",
        "users": []string{"Alice", "Bob", "Charlie"},
    }
    c.Render("index.html", data)
})

app.RunDev(":8080") // Hot reload in development

WebSocket Chat:

app.Get("/ws", func(c *smallapi.Context) {
    c.Upgrade(func(ws *smallapi.WebSocket) {
        for {
            message, err := ws.ReadMessage()
            if err != nil {
                break
            }
            ws.WriteMessage([]byte("Echo: " + string(message)))
        }
    })
})

Authentication:

app.Use(smallapi.Session())

app.Post("/login", func(c *smallapi.Context) {
    // ... validate credentials ...
    c.Session().Set("user_id", userID)
    c.JSON(map[string]string{"status": "logged in"})
})

// Protected routes
api := app.Group("/api")
api.Use(smallapi.RequireAuth())
api.Get("/profile", func(c *smallapi.Context) {
    userID := c.Session().Get("user_id")
    c.JSON(map[string]interface{}{"user_id": userID})
})
